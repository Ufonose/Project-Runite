-- ServerScriptService/Items/OreCrusher.lua (or wherever your machine scripts are)

local Players = game:GetService("Players")
local RepStorage = game:GetService("ReplicatedStorage")
local CollectionService = game:GetService("CollectionService")

local ITEM_HANDLER = require(RepStorage.Modules.ItemHandler)
local ITEM_INFO = require(script.Parent.Parent.ItemStats)
local PLOT = script.Parent.Parent.Parent.Parent

local CRUSH_INPUT = script.Parent:WaitForChild("CrushInput")
local CRUSH_OUTPUT = script.Parent:WaitForChild("CrushOutput")

-- Settings
local SPREAD_WIDTH = 2.7 
local DETECT_INTERVAL = 0.1 -- How often to check for ores (10 times a second)

-- Track recently crushed ores to prevent duplicates
local crushedOres = {}

local ACCEPTED_RAW_ORES = {
	"Iron Ore",
	"Gold Ore",
	"Copper Ore",
}

local function isAcceptedOre(oreModelName)
	for _, acceptedName in ipairs(ACCEPTED_RAW_ORES) do
		if oreModelName == acceptedName then return true end
	end
	return false
end

local function crushOre(ORE)
	-- 1. VALIDATION
	if not ORE or crushedOres[ORE] then return end
	crushedOres[ORE] = true

	local oreModelTag = ORE:FindFirstChild("OreModelName")
	if not oreModelTag then
		ORE:Destroy()
		return
	end

	local originalModelName = oreModelTag.Value

	-- Skip if not raw or already crushed
	if not isAcceptedOre(originalModelName) or string.find(originalModelName, "_Crushed") then
		ORE:Destroy()
		return
	end

	-- 2. PREPARE REFINED MODEL
	local oreFolder = RepStorage:FindFirstChild("OreModels")
	local refinedModelName = originalModelName .. "_Crushed"
	local refinedModel = oreFolder and oreFolder:FindFirstChild(refinedModelName)

	if not refinedModel then
		warn("Refined model '" .. refinedModelName .. "' not found!")
		ORE:Destroy()
		return
	end

	-- 3. CLONE AND SETUP
	local newOre = refinedModel:Clone()
	newOre.Name = ORE.Name

	-- Add Tags & Attributes so the LocalConveyorManager takes control
	CollectionService:AddTag(newOre, "MovingOre")
	newOre:SetAttribute("OreID", math.random(1, 1000))
	newOre:SetAttribute("LaneOffset", (math.random(-80, 80) / 100))

	-- StringValue tag for the next machine
	local newOreModelTag = Instance.new("StringValue")
	newOreModelTag.Name = "OreModelName"
	newOreModelTag.Value = refinedModelName
	newOreModelTag.Parent = newOre

	-- Copy Cash and upgrade values
	if ORE:FindFirstChild("Cash") then
		local cashValue = Instance.new("NumberValue")
		cashValue.Name = "Cash"
		cashValue.Value = ORE.Cash.Value
		cashValue.Parent = newOre
	end

	for _, child in pairs(ORE:GetChildren()) do
		if child:IsA("NumberValue") and child.Name ~= "Cash" then
			child:Clone().Parent = newOre
		end
	end

	-- 4. POSITIONING & PHYSICS
	local randomX = math.random(-SPREAD_WIDTH/2 * 100, SPREAD_WIDTH/2 * 100) / 100

	if newOre:IsA("Model") then
		newOre:PivotTo(CRUSH_OUTPUT.CFrame * CFrame.new(randomX, 0, 0))
		if newOre.PrimaryPart then newOre.PrimaryPart.Anchored = false end
	else
		newOre.CFrame = CRUSH_OUTPUT.CFrame * CFrame.new(randomX, 0, 0)
		newOre.Anchored = false
	end

	newOre.Parent = ORE.Parent

	-- Network Ownership
	local plotOwner = Players:FindFirstChild(PLOT.Owner.Value)
	if plotOwner then
		local targetPart = newOre:IsA("Model") and newOre.PrimaryPart or newOre
		if targetPart and targetPart:IsA("BasePart") then
			targetPart:SetNetworkOwner(plotOwner)
		end
	end

	-- 5. CLEANUP
	local oldOreRef = ORE
	ORE:Destroy()

	task.delay(1, function()
		crushedOres[oldOreRef] = nil
	end)
end

-- ============================================================
-- THE DETECTION LOOP (Replaces .Touched)
-- ============================================================
task.spawn(function()
	local overlapParams = OverlapParams.new()
	overlapParams.FilterType = Enum.RaycastFilterType.Include

	while true do
		task.wait(DETECT_INTERVAL)

		-- Update filter to only look for currently existing ores
		overlapParams.FilterDescendantsInstances = CollectionService:GetTagged("MovingOre")

		local partsInInput = workspace:GetPartBoundsInBox(CRUSH_INPUT.CFrame, CRUSH_INPUT.Size, overlapParams)

		for _, hit in pairs(partsInInput) do
			local ORE = nil

			-- Identify if hit is part of an ore model or the part itself
			if hit:FindFirstChild("Cash") then
				ORE = hit
			elseif hit.Parent and hit.Parent:FindFirstChild("Cash") then
				ORE = hit.Parent
			end

			if ORE and not crushedOres[ORE] then
				-- Apply Upgrades
				ITEM_HANDLER.UPGRADE_ORE(
					PLOT, 
					script.Parent.Parent, 
					ORE, 
					ITEM_INFO.Addition, 
					ITEM_INFO.Multiplier, 
					ITEM_INFO.ItemId, 
					30
				)
				-- Process the crush
				crushOre(ORE)
			end
		end
	end
end)

print("Ore Crusher Logic Loaded (Active Scanning Mode)")
