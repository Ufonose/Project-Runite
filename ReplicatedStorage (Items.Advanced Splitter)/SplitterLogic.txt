-- ServerScriptService/Items/ItemSplitter.lua
local Players = game:GetService("Players")
local CollectionService = game:GetService("CollectionService")

local ITEM_INFO = require(script.Parent.ItemStats)
local PLOT = script.Parent.Parent.Parent

local SPLITTER = script.Parent
local SPLIT_DETECTOR = SPLITTER:WaitForChild("SortDetector")

-- Settings
local DETECT_INTERVAL = 0.05 
local DETECTION_HEIGHT = 4
local DETECTION_OFFSET = -1 

-- The global toggle (flips every time a NEW ore is assigned)
local nextDirectionLeft = true

-- Calculate split velocity
local function getSplitVelocity(goLeft)
	local baseCFrame = SPLIT_DETECTOR.CFrame
	local speed = ITEM_INFO.SortSpeed or 12 -- Slightly faster to ensure clearing
	local angleDeg = ITEM_INFO.SortAngle or 45

	local angle = math.rad(goLeft and angleDeg or -angleDeg)
	return (baseCFrame * CFrame.Angles(0, angle, 0)).LookVector * speed
end

-- ============================================================
-- THE DETECTION & SPLIT LOOP
-- ============================================================
task.spawn(function()
	local overlapParams = OverlapParams.new()
	overlapParams.FilterType = Enum.RaycastFilterType.Exclude
	overlapParams.FilterDescendantsInstances = {SPLITTER}

	while true do
		task.wait(DETECT_INTERVAL)

		local detectionSize = Vector3.new(SPLIT_DETECTOR.Size.X, DETECTION_HEIGHT, SPLIT_DETECTOR.Size.Z)
		local detectionCFrame = SPLIT_DETECTOR.CFrame * CFrame.new(0, (DETECTION_HEIGHT/2) + DETECTION_OFFSET, 0)

		local partsInRegion = workspace:GetPartBoundsInBox(detectionCFrame, detectionSize, overlapParams)

		for _, hit in pairs(partsInRegion) do
			local ITEM = nil

			if hit:FindFirstChild("Cash") then
				ITEM = hit
			elseif hit.Parent and hit.Parent:FindFirstChild("Cash") then
				ITEM = hit.Parent
			end

			if ITEM then
				-- 1. ASSIGN DIRECTION (If it doesn't have one yet)
				local assignedDir = ITEM:GetAttribute("SplitDir")

				if assignedDir == nil then
					-- This is a brand new ore! 
					-- Assign the current global direction and then flip the global toggle
					assignedDir = nextDirectionLeft and "Left" or "Right"
					ITEM:SetAttribute("SplitDir", assignedDir)

					nextDirectionLeft = not nextDirectionLeft
					-- print("New Ore Assigned to: " .. assignedDir)
				end

				-- 2. APPLY PHYSICS BASED ON ASSIGNED DIRECTION
				local isLeft = (assignedDir == "Left")
				local velocity = getSplitVelocity(isLeft)

				local targetPart = (ITEM:IsA("Model") and ITEM.PrimaryPart) or ITEM
				if targetPart and targetPart:IsA("BasePart") then
					targetPart.AssemblyLinearVelocity = velocity

					-- 3. NETWORK OWNERSHIP
					local plotOwnerName = PLOT:FindFirstChild("Owner") and PLOT.Owner.Value
					if plotOwnerName and plotOwnerName ~= "" then
						local plotOwner = Players:FindFirstChild(plotOwnerName)
						if plotOwner and targetPart:GetNetworkOwner() ~= plotOwner then
							pcall(function() targetPart:SetNetworkOwner(plotOwner) end)
						end
					end
				end
			end
		end
	end
end)
