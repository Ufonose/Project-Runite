local Players = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")

local player = Players.LocalPlayer

-- Track player location (Cave / Surface)
local locationValue = Instance.new("StringValue")
locationValue.Name = "PlayerLocation"
locationValue.Value = "Cave"
locationValue.Parent = player

-- Wait for plot to be assigned first
local setPlot = player:WaitForChild("SetPlot")
local plotName = setPlot.Value and setPlot.Value.Name

-- ── Cave elevator prompts (go to surface) ────────────────────────────────
-- Connect ALL cave elevators, not just the player's own, so they can
-- return to the surface from any plot's cave.
local caveElevators = workspace:WaitForChild("CaveElevators")

local function connectCavePrompt(cavePart)
	local prompt = cavePart:WaitForChild("ProximityPrompt", 5)
	if prompt then
		prompt.Triggered:Connect(function()
			locationValue.Value = "Surface"
			ReplicatedStorage.Events.TeleportToSurface:FireServer(cavePart.Name)
		end)
	end
end

for _, cavePart in pairs(caveElevators:GetChildren()) do
	connectCavePrompt(cavePart)
end
caveElevators.ChildAdded:Connect(connectCavePrompt)

-- ── Surface elevator prompts (return to cave) ────────────────────────────
local surfaceElevators = workspace:WaitForChild("SurfaceElevators")

local function connectSurfacePrompt(surfacePart)
	local prompt = surfacePart:WaitForChild("ProximityPrompt", 5)
	if prompt then
		prompt.Triggered:Connect(function()
			locationValue.Value = "Cave"
			ReplicatedStorage.Events.TeleportToCave:FireServer(surfacePart.Name)
		end)
	end
end

for _, surfacePart in pairs(surfaceElevators:GetChildren()) do
	connectSurfacePrompt(surfacePart)
end
surfaceElevators.ChildAdded:Connect(connectSurfacePrompt)
