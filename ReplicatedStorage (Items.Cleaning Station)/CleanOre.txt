-- ServerScriptService/Items/OreCleaner.lua

local Players = game:GetService("Players")
local RepStorage = game:GetService("ReplicatedStorage")
local CollectionService = game:GetService("CollectionService")

local ITEM_HANDLER = require(RepStorage.Modules.ItemHandler)
local ITEM_INFO = require(script.Parent.Parent.ItemStats)
local PLOT = script.Parent.Parent.Parent.Parent

local CLEAN_INPUT = script.Parent:WaitForChild("CleanInput")
local CLEAN_OUTPUT = script.Parent:WaitForChild("CleanOutput")

local SPREAD_WIDTH = 2.7 
local DETECT_INTERVAL = 0.1 

local cleanedOres = {}

local function cleanOre(ORE)
	if not ORE or cleanedOres[ORE] then return end

	local oreModelTag = ORE:FindFirstChild("OreModelName")
	if not oreModelTag then return end

	local originalModelName = oreModelTag.Value

	if not string.find(originalModelName, "_Crushed") then
		return
	end

	cleanedOres[ORE] = true

	ITEM_HANDLER.UPGRADE_ORE(
		PLOT, 
		script.Parent.Parent, 
		ORE, 
		ITEM_INFO.Addition, 
		ITEM_INFO.Multiplier, 
		ITEM_INFO.ItemId, 
		30
	)

	local oreFolder = RepStorage:FindFirstChild("OreModels")
	local refinedModelName = string.gsub(originalModelName, "_Crushed", "_Cleaned")
	local refinedModel = oreFolder and oreFolder:FindFirstChild(refinedModelName)

	if not refinedModel then
		warn("Cleaned model '" .. refinedModelName .. "' not found!")
		cleanedOres[ORE] = nil
		return
	end

	local newOre = refinedModel:Clone()
	newOre.Name = ORE.Name
	CollectionService:AddTag(newOre, "MovingOre")
	newOre:SetAttribute("OreID", math.random(1, 1000000))
	newOre:SetAttribute("LaneOffset", (math.random(-80, 80) / 100))

	local newOreModelTag = Instance.new("StringValue")
	newOreModelTag.Name = "OreModelName"
	newOreModelTag.Value = refinedModelName
	newOreModelTag.Parent = newOre

	if ORE:FindFirstChild("Cash") then
		local cashValue = Instance.new("NumberValue")
		cashValue.Name = "Cash"
		cashValue.Value = ORE.Cash.Value
		cashValue.Parent = newOre
	end

	for _, child in pairs(ORE:GetChildren()) do
		if child:IsA("NumberValue") and child.Name ~= "Cash" then
			child:Clone().Parent = newOre
		end
	end

	local randomX = math.random(-SPREAD_WIDTH/2 * 100, SPREAD_WIDTH/2 * 100) / 100

	if newOre:IsA("Model") then
		newOre:PivotTo(CLEAN_OUTPUT.CFrame * CFrame.new(randomX, 0, 0))
		if newOre.PrimaryPart then newOre.PrimaryPart.Anchored = false end
	else
		newOre.CFrame = CLEAN_OUTPUT.CFrame * CFrame.new(randomX, 0, 0)
		newOre.Anchored = false
	end

	for _, part in ipairs(newOre:GetDescendants()) do
		if part:IsA("BasePart") then part.CollisionGroup = "Ores" end
	end
	if newOre:IsA("BasePart") then newOre.CollisionGroup = "Ores" end

	newOre.Parent = ORE.Parent

	local plotOwner = Players:FindFirstChild(PLOT.Owner.Value)
	if plotOwner then
		local targetPart = newOre:IsA("Model") and newOre.PrimaryPart or newOre
		if targetPart and targetPart:IsA("BasePart") then
			targetPart:SetNetworkOwner(plotOwner)
		end
	end

	local oldOreRef = ORE
	ORE:Destroy()

	task.delay(1, function()
		cleanedOres[oldOreRef] = nil
	end)
end

-- ============================================================
-- THE DETECTION LOOP
-- ============================================================
task.spawn(function()
	local overlapParams = OverlapParams.new()
	overlapParams.FilterType = Enum.RaycastFilterType.Include

	while true do
		task.wait(DETECT_INTERVAL)
		overlapParams.FilterDescendantsInstances = CollectionService:GetTagged("MovingOre")

		local partsInInput = workspace:GetPartBoundsInBox(CLEAN_INPUT.CFrame, CLEAN_INPUT.Size, overlapParams)

		for _, hit in pairs(partsInInput) do
			local ORE = nil
			if hit:FindFirstChild("Cash") then
				ORE = hit
			elseif hit.Parent and hit.Parent:FindFirstChild("Cash") then
				ORE = hit.Parent
			end

			-- Just send it to the function; the function now decides if it's worth upgrading/cleaning
			if ORE then
				cleanOre(ORE)
			end
		end
	end
end)
