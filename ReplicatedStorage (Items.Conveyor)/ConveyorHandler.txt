local RepStorage = game:GetService("ReplicatedStorage")
local RunService = game:GetService("RunService")
local ITEM_HANDLER = require(RepStorage.Modules.ItemHandler)

-- Wait for ConvVelo to be created (it's created by ItemHandler.txt on startup)
local ConvVelo = script.Parent:WaitForChild("ConvVelo", 5)

if not ConvVelo then
	warn("ConvVelo not found on", script.Parent:GetFullName())
	return
end

-- Continuously move the conveyor using the ConvVelo value
RunService.Heartbeat:Connect(function()
	ITEM_HANDLER.CONVEYOR_MOVE(script.Parent, ConvVelo.Value)
end)
