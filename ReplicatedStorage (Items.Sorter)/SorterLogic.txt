-- Get sort direction for an ore type
local function getSortDirection(oreModelName)
	local config = SORT_CONFIG:FindFirstChild(oreModelName)
	if config and config:IsA("StringValue") then
		return config.Value -- "Left", "Right", or "Straight"
	end
	return "Straight"
end

-- Calculate sort velocity
local function getSortVelocity(direction)
	local baseCFrame = SORT_DETECTOR.CFrame
	local speed = ITEM_INFO.SortSpeed or 10
	local angleDeg = ITEM_INFO.SortAngle or 45

	if direction == "Left" then
		local angle = math.rad(angleDeg)
		return (baseCFrame * CFrame.Angles(0, angle, 0)).LookVector * speed
	elseif direction == "Right" then
		local angle = math.rad(-angleDeg)
		return (baseCFrame * CFrame.Angles(0, angle, 0)).LookVector * speed
	else -- "Straight"
		return baseCFrame.LookVector * speed
	end
end

-- FIX: Connect the Changed event to newly added configs
local function connectConfigChanged(config)
	if config:IsA("StringValue") then
		config.Changed:Connect(function()
			--print("Config changed:", config.Name, "->", config.Value)
			updateDisplay()
		end)
	end
end

-- Update display when config changes
SORT_CONFIG.ChildAdded:Connect(function(child)
	connectConfigChanged(child)  -- FIX: Connect Changed event to new configs
	updateDisplay()
end)

SORT_CONFIG.ChildRemoved:Connect(updateDisplay)

-- Connect to existing configs
for _, config in pairs(SORT_CONFIG:GetChildren()) do
	connectConfigChanged(config)
end

-- ============================================================
-- THE DETECTION & MOVEMENT LOOP
-- ============================================================
task.spawn(function()
	local overlapParams = OverlapParams.new()
	overlapParams.FilterType = Enum.RaycastFilterType.Exclude
	overlapParams.FilterDescendantsInstances = {SORTER}

	while true do
		task.wait(DETECT_INTERVAL)

		-- 1. DEFINE DETECTION BOX
		local detectionSize = Vector3.new(SORT_DETECTOR.Size.X, DETECTION_HEIGHT, SORT_DETECTOR.Size.Z)
		local detectionCFrame = SORT_DETECTOR.CFrame * CFrame.new(0, (DETECTION_HEIGHT/2) + DETECTION_OFFSET, 0)

		-- 2. SCAN FOR ITEMS
		local partsInRegion = workspace:GetPartBoundsInBox(detectionCFrame, detectionSize, overlapParams)

		-- Use a set to prevent processing the same model multiple times in one tick
		local processedThisTick = {}

		for _, hit in pairs(partsInRegion) do
			local ITEM = nil
			local modelName = ""

			-- Identify the item and its model name
			if hit:FindFirstChild("Cash") then
				ITEM = hit
			elseif hit.Parent and hit.Parent:FindFirstChild("Cash") then
				ITEM = hit.Parent
			end

			if ITEM and not processedThisTick[ITEM] then
				processedThisTick[ITEM] = true

				-- Get the tag name (OreModelName is our standard now)
				local nameTag = ITEM:FindFirstChild("OreModelName") or ITEM:FindFirstChild("IngotType")
				modelName = nameTag and nameTag.Value or ITEM.Name

				-- 3. APPLY DIRECTIONAL VELOCITY
				local direction = getSortDirection(modelName)
				local velocity = getSortVelocity(direction)

				-- Apply physics to the PrimaryPart or the BasePart itself
				local targetPart = (ITEM:IsA("Model") and ITEM.PrimaryPart) or ITEM
				if targetPart and targetPart:IsA("BasePart") then
					targetPart.AssemblyLinearVelocity = velocity

					-- 4. ENSURE NETWORK OWNERSHIP
					-- This is vital so the player's client handles the velocity smoothly
					local plotOwnerName = PLOT:FindFirstChild("Owner") and PLOT.Owner.Value
					if plotOwnerName ~= "" then
						local plotOwner = Players:FindFirstChild(plotOwnerName)
						if plotOwner then
							local success, _ = targetPart:CanSetNetworkOwnership()
							if success then
								-- Only set if it's not already owned (prevents overhead)
								if targetPart:GetNetworkOwner() ~= plotOwner then
									targetPart:SetNetworkOwner(plotOwner)
								end
							end
						end
					end
				end
			end
		end
	end
end)

-- Initial GUI update call
