-- ServerScriptService/Items/ItemSorter.lua
local Players = game:GetService("Players")
local RepStorage = game:GetService("ReplicatedStorage")
local CollectionService = game:GetService("CollectionService")

local ITEM_INFO = require(script.Parent.ItemStats)
local PLOT = script.Parent.Parent.Parent

local SORTER = script.Parent
local SORT_DETECTOR = SORTER:WaitForChild("SortDetector")

-- Storage for sort configuration
if not SORTER:FindFirstChild("SortConfig") then
	local config = Instance.new("Folder")
	config.Name = "SortConfig"
	config.Parent = SORTER
end
local SORT_CONFIG = SORTER.SortConfig

-- Settings
local DETECT_INTERVAL  = 0.05
local DETECTION_HEIGHT = 4
local DETECTION_OFFSET = -1

-- Ores that have already been kicked this pass through the SortDetector.
-- Cleared once they leave the zone so they can be kicked again on re-entry.
local oresInZone = {}

-- Get sort direction for an ore type
local function getSortDirection(oreModelName)
	local config = SORT_CONFIG:FindFirstChild(oreModelName)
	if config and config:IsA("StringValue") then
		return config.Value -- "Left", "Right", or "Straight"
	end
	return "Straight"
end


-- Teleport an item to a target part's position and unanchor it
-- so it falls naturally onto the conveyor below
local function teleportItem(ITEM, targetPart)
	local plotOwnerName = PLOT:FindFirstChild("Owner") and PLOT.Owner.Value
	local plotOwner = plotOwnerName and plotOwnerName ~= "" and Players:FindFirstChild(plotOwnerName)

	local clone = ITEM:Clone()

	local spawnCFrame = targetPart.CFrame * CFrame.new(0, 2, 0)
	if clone:IsA("Model") then
		clone:PivotTo(spawnCFrame)
	else
		clone.CFrame = spawnCFrame
	end

	-- Parent FIRST so network ownership can be set
	clone.Parent = ITEM.Parent

	local clonePart = (clone:IsA("Model") and clone.PrimaryPart) or clone
	if clonePart and clonePart:IsA("BasePart") then
		clonePart.Anchored = false
		clonePart.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
		clonePart.AssemblyAngularVelocity = Vector3.new(0, 0, 0)
		if plotOwner then
			pcall(function() clonePart:SetNetworkOwner(plotOwner) end)
		end
	end

	-- Tag AFTER parenting so CollectionService signals fire correctly
	CollectionService:AddTag(clone, "MovingOre")

	-- Immediately mark the clone as already processed so the sorter
	-- doesn't detect it again if it spawns inside the SortDetector zone
	oresInZone[clone] = true

	-- Destroy the original
	CollectionService:RemoveTag(ITEM, "MovingOre")
	ITEM:Destroy()

	return clone
end

-- Scan a region and return deduplicated items inside it
local function getItemsInRegion(regionCFrame, regionSize, overlapParams)
	local parts = workspace:GetPartBoundsInBox(regionCFrame, regionSize, overlapParams)
	local seen, items = {}, {}

	for _, hit in pairs(parts) do
		local item = nil
		if hit:FindFirstChild("Cash") then
			item = hit
		elseif hit.Parent and hit.Parent:FindFirstChild("Cash") then
			item = hit.Parent
		end
		if item and not seen[item] then
			seen[item] = true
			table.insert(items, item)
		end
	end

	return items
end

		if #items == 0 then
			local emptyLabel = Instance.new("TextLabel")
			emptyLabel.Size = UDim2.new(1, 0, 0, 15)
			emptyLabel.BackgroundTransparency = 1
			emptyLabel.Text = "(None)"
			emptyLabel.TextColor3 = Color3.fromRGB(100, 100, 100)
			emptyLabel.TextSize = 7
			emptyLabel.Font = Enum.Font.Code
			emptyLabel.Parent = scroll
		else
			for _, oreName in ipairs(items) do
				local oreLabel = Instance.new("TextLabel")
				oreLabel.Size = UDim2.new(1, -10, 0, 15)
				oreLabel.BackgroundColor3 = Color3.fromRGB(25, 25, 30)
				oreLabel.BorderSizePixel = 0
				oreLabel.Text = "  " .. oreName
				oreLabel.TextColor3 = textColor
				oreLabel.TextSize = 7
				oreLabel.Font = Enum.Font.Code
				oreLabel.TextXAlignment = Enum.TextXAlignment.Left
				oreLabel.Parent = scroll
				local corner = Instance.new("UICorner")
				corner.CornerRadius = UDim.new(0, 3)
				corner.Parent = oreLabel
			end
		end
	end

	local leftItems, rightItems = {}, {}
	for _, config in pairs(SORT_CONFIG:GetChildren()) do
		if config:IsA("StringValue") then
			local oreName = config.Name:gsub("OreModel", ""):gsub("_", " ")
			if config.Value == "Left" then
				table.insert(leftItems, oreName)
			elseif config.Value == "Right" then
				table.insert(rightItems, oreName)
			end
		end
	end

	table.sort(leftItems)
	table.sort(rightItems)

	populateScroll(leftScroll,  leftItems,  Color3.fromRGB(255, 200, 200))
	populateScroll(rightScroll, rightItems, Color3.fromRGB(200, 200, 255))

	leftScroll.CanvasSize  = UDim2.new(0, 0, 0, leftLayout.AbsoluteContentSize.Y  + 10)
	rightScroll.CanvasSize = UDim2.new(0, 0, 0, rightLayout.AbsoluteContentSize.Y + 10)
end

local function connectConfigChanged(config)
	if config:IsA("StringValue") then
		config.Changed:Connect(updateDisplay)
	end
end

SORT_CONFIG.ChildAdded:Connect(function(child)
	connectConfigChanged(child)
	updateDisplay()
end)
SORT_CONFIG.ChildRemoved:Connect(updateDisplay)
for _, config in pairs(SORT_CONFIG:GetChildren()) do
	connectConfigChanged(config)
end

-- ============================================================
-- THE DETECTION & MOVEMENT LOOP
-- ============================================================
task.spawn(function()
	local overlapParams = OverlapParams.new()
	overlapParams.FilterType = Enum.RaycastFilterType.Exclude
	overlapParams.FilterDescendantsInstances = {SORTER}

	while true do
		task.wait(DETECT_INTERVAL)

		local sortSize   = Vector3.new(SORT_DETECTOR.Size.X, DETECTION_HEIGHT, SORT_DETECTOR.Size.Z)
		local sortCFrame = SORT_DETECTOR.CFrame * CFrame.new(0, (DETECTION_HEIGHT / 2) + DETECTION_OFFSET, 0)
		local sortItems  = getItemsInRegion(sortCFrame, sortSize, overlapParams)

		local currentOresInZone = {}

		for _, ITEM in ipairs(sortItems) do
			currentOresInZone[ITEM] = true

			if not oresInZone[ITEM] then
				oresInZone[ITEM] = true

				local nameTag   = ITEM:FindFirstChild("OreModelName") or ITEM:FindFirstChild("IngotType")
				local modelName = nameTag and nameTag.Value or ITEM.Name
				local direction = getSortDirection(modelName)

				-- Find the spawn part for this direction
				local spawnPartName = direction == "Left" and "LeftSort"
					or direction == "Right" and "RightSort"
					or "ForwardSort"

				local spawnPart = SORTER:FindFirstChild(spawnPartName)
				if spawnPart then
					teleportItem(ITEM, spawnPart)
				end
				-- If no spawn part found, item just continues through untouched
			end
		end

		-- Clear ores that have left the sort zone
		for ore in pairs(oresInZone) do
			if not currentOresInZone[ore] then
				oresInZone[ore] = nil
			end
		end
	end
end)

updateDisplay()

print("Sorter logic loaded!")
