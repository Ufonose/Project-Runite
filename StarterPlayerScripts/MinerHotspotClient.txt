local Players = game:GetService("Players")
local RepStorage = game:GetService("ReplicatedStorage")
local TweenService = game:GetService("TweenService")
local player = Players.LocalPlayer
local Camera = workspace.CurrentCamera
local SoundManager = require(RepStorage.Modules.SoundManager)
local SpeedrunReset = RepStorage:WaitForChild("Events"):WaitForChild("SpeedrunReset")

local dataLoaded = player:WaitForChild("DataLoaded", 30)
if not dataLoaded then
	warn("DataLoaded tag not found - player data didn't load!")
	return
end

player:WaitForChild("SetPlot", 10)
local plot = player.SetPlot.Value
if not plot then
	warn("SetPlot.Value is nil!")
	return
end

local hotspotFolder = plot:WaitForChild("MinerHotspots", 10)
if not hotspotFolder then
	warn("No MinerHotspots folder found in plot!")
	return
end

-- Stores resetCostDisplay callbacks for research-locked hotspots so they can
-- be refreshed live when the player finishes all research
local researchLockedResetCallbacks = {}

local function createModernHotspotUI(parent, info)
	local billboardGui = Instance.new("BillboardGui")
	billboardGui.Name = "BuyPrompt"
	billboardGui.StudsOffset = Vector3.new(0, 5, 0)
	billboardGui.AlwaysOnTop = false
	billboardGui.MaxDistance = 50
	billboardGui.Adornee = parent
	billboardGui.Parent = parent

	local function updateSize()
		local sw = Camera.ViewportSize.X
		if sw < 900 then
			billboardGui.Size = UDim2.new(0, 130, 0, 56)
		else
			billboardGui.Size = UDim2.new(0, 180, 0, 78)
		end
	end

	updateSize()
	Camera:GetPropertyChangedSignal("ViewportSize"):Connect(updateSize)

	local container = Instance.new("Frame")
	container.Size = UDim2.new(1, 0, 1, 0)
	container.BackgroundColor3 = Color3.fromRGB(25, 25, 30)
	container.BackgroundTransparency = 0.1
	container.BorderSizePixel = 0
	container.ZIndex = 10
	container.Parent = billboardGui

	local containerCorner = Instance.new("UICorner")
	containerCorner.CornerRadius = UDim.new(0, 12)
	containerCorner.Parent = container

	local blur = Instance.new("Frame")
	blur.Size = UDim2.new(1, 0, 1, 0)
	blur.BackgroundColor3 = Color3.fromRGB(15, 15, 20)
	blur.BackgroundTransparency = 0.3
	blur.BorderSizePixel = 0
	blur.ZIndex = 11
	blur.Parent = container

	local blurCorner = Instance.new("UICorner")
	blurCorner.CornerRadius = UDim.new(0, 12)
	blurCorner.Parent = blur

	local stroke = Instance.new("UIStroke")
	stroke.Color = Color3.fromRGB(60, 60, 80)
	stroke.Thickness = 2
	stroke.Transparency = 0.5
	stroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
	stroke.Parent = container

	local content = Instance.new("Frame")
	content.Size = UDim2.new(0.92, 0, 0.81, 0)
	content.Position = UDim2.new(0.04, 0, 0.094, 0)
	content.BackgroundTransparency = 1
	content.ZIndex = 12
	content.Parent = container

	local titleLabel = Instance.new("TextLabel")
	titleLabel.Name = "TitleLabel"
	titleLabel.Size = UDim2.new(1, 0, 0.29, 0)
	titleLabel.Position = UDim2.new(0, 0, 0, 0)
	titleLabel.BackgroundTransparency = 1
	local displayName = info.Config.MinerName:gsub("%s*%d+$", "")
	titleLabel.Text = displayName
	titleLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
	titleLabel.TextScaled = false
	titleLabel.TextSize = 16
	titleLabel.Font = Enum.Font.GothamBold
	titleLabel.TextXAlignment = Enum.TextXAlignment.Center
	titleLabel.ZIndex = 13
	titleLabel.Parent = content

	local divider = Instance.new("Frame")
	divider.Size = UDim2.new(0.95, 0, 0.015, 0)
	divider.Position = UDim2.new(0.025, 0, 0.35, 0)
	divider.BackgroundColor3 = Color3.fromRGB(60, 60, 80)
	divider.BackgroundTransparency = 0.5
	divider.BorderSizePixel = 0
	divider.ZIndex = 12
	divider.Parent = content

	local costContainer = Instance.new("Frame")
	costContainer.Size = UDim2.new(0.95, 0, 0.46, 0)
	costContainer.Position = UDim2.new(0.025, 0, 0.43, 0)
	costContainer.BackgroundColor3 = Color3.fromRGB(50, 55, 65)
	costContainer.BorderSizePixel = 0
	costContainer.ZIndex = 12
	costContainer.Parent = content

	local costCorner = Instance.new("UICorner")
	costCorner.CornerRadius = UDim.new(0, 8)
	costCorner.Parent = costContainer

	local costStroke = Instance.new("UIStroke")
	costStroke.Color = Color3.fromRGB(0, 0, 0)
	costStroke.Thickness = 1
	costStroke.Transparency = 0.7
	costStroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
	costStroke.Parent = costContainer

	local costLabel = Instance.new("TextLabel")
	costLabel.Name = "CostLabel"
	costLabel.Size = UDim2.new(0.92, 0, 1, 0)
	costLabel.Position = UDim2.new(0.04, 0, 0, 0)
	costLabel.BackgroundTransparency = 1
	costLabel.ZIndex = 13
	costLabel.Font = Enum.Font.GothamBold
	costLabel.TextScaled = false
	costLabel.TextSize = 15
	costLabel.TextXAlignment = Enum.TextXAlignment.Center
	costLabel.Parent = costContainer

	local function resetCostDisplay()
		if info.IsResearchLocked then
			costLabel.RichText = true
			costLabel.Text = "🔒 Complete Ages 1, 2 & 3\n<font color='#00cc66'>⚡ Instant Buy: \u{E002}50</font>"
			costLabel.TextColor3 = Color3.fromRGB(200, 150, 80)
			costLabel.TextSize = Camera.ViewportSize.X < 900 and 9 or 13
		elseif info.Config.CashCost then
			local cost = (info.InSpeedrun and info.Config.SpeedrunCashCost) or info.Config.CashCost
			costLabel.Text = "$" .. string.format("%d", cost / 1000) .. "K"
			costLabel.TextColor3 = Color3.fromRGB(100, 220, 100)
			costLabel.TextSize = 15
		elseif info.Config.Cost == 0 then
			costLabel.Text = "FREE!"
			costLabel.TextColor3 = Color3.fromRGB(100, 255, 100)
		else
			costLabel.Text = info.Config.Cost .. " Tokens"
			costLabel.TextColor3 = Color3.fromRGB(134, 207, 255)
			costLabel.TextSize = 15 
		end
	end
	resetCostDisplay()

	local pulseInfo = TweenInfo.new(1.5, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut, -1, true)
	local pulseTween = TweenService:Create(costStroke, pulseInfo, { Transparency = 0.4 })
	pulseTween:Play()

	return billboardGui, costLabel, costContainer, pulseTween, resetCostDisplay
end

local function setupHotspot(hotspot)
	local hotspotId = hotspot:WaitForChild("HotspotId", 5)
	if not hotspotId then
		warn("No HotspotId in", hotspot.Name)
		return
	end

	if not RepStorage.Events:FindFirstChild("GetHotspotInfo") then
		warn("GetHotspotInfo RemoteFunction not found!")
		return
	end

	local success, info = pcall(function()
		return RepStorage.Events.GetHotspotInfo:InvokeServer(hotspotId.Value)
	end)

	if not success then
		warn("Error calling GetHotspotInfo:", info)
		return
	end

	if not info or not info.Config then
		warn("No config returned for hotspot", hotspotId.Value)
		return
	end

	if info.IsUnlocked then return end

	local buyButton = hotspot:WaitForChild("BuyButton", 10)
	if not buyButton then
		warn("No BuyButton in", hotspot.Name, "- hotspot is locked but has no BuyButton part")
		return
	end

	local ghostMiner = nil
	local minerModel = RepStorage.Items:FindFirstChild(info.Config.MinerName)
	if minerModel then
		ghostMiner = minerModel:Clone()
		ghostMiner.Name = "GhostMiner"

		for _, descendant in pairs(ghostMiner:GetDescendants()) do
			if descendant:IsA("BasePart") then
				descendant.Transparency = descendant.Name == "Hitbox" and 1 or 0.7
				descendant.CanCollide = false
				descendant.Anchored = true
			end
		end

		local positionPart = hotspot:WaitForChild("Position", 5)
		if positionPart then
			ghostMiner:SetPrimaryPartCFrame(positionPart.CFrame)
		end

		ghostMiner.Parent = hotspot
	else
		warn("Miner model not found:", info.Config.MinerName)
	end

	local billboardGui, costLabel, costContainer, pulseTween, resetCostDisplay = createModernHotspotUI(buyButton, info)

	-- Register callback so AllResearchCompleted can refresh this label live
	if info.IsResearchLocked then
		researchLockedResetCallbacks[hotspotId.Value] = function()
			info.IsResearchLocked = false
			resetCostDisplay()
		end
	end

	local proximityPrompt = buyButton:FindFirstChild("UnlockPrompt")
	if not proximityPrompt then
		proximityPrompt = Instance.new("ProximityPrompt")
		proximityPrompt.Name = "UnlockPrompt"
		proximityPrompt.ActionText = "Unlock Miner"
		proximityPrompt.HoldDuration = 0.5
		proximityPrompt.KeyboardKeyCode = Enum.KeyCode.E
		proximityPrompt.MaxActivationDistance = 10
		proximityPrompt.RequiresLineOfSight = false
		proximityPrompt.Parent = buyButton
	end

	proximityPrompt.Triggered:Connect(function(playerWhoTriggered)
		if playerWhoTriggered ~= player then return end

		pulseTween:Pause()

		local ok, result, errorMsg = pcall(function()
			return RepStorage.Events.UnlockMinerHotspot:InvokeServer(hotspotId.Value)
		end)

		if not ok then
			warn("Error calling UnlockMinerHotspot:", result)
			costLabel.Text = "❌ Error!"
			costLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
			task.wait(1.5)
			resetCostDisplay()
			pulseTween:Play()
			return
		end

		if result == true then
			costLabel.Text = "✓ Unlocked!"
			costLabel.TextColor3 = Color3.fromRGB(100, 255, 100)

			local fadeInfo = TweenInfo.new(0.5, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
			TweenService:Create(billboardGui.Frame, fadeInfo, { BackgroundTransparency = 1 }):Play()
			task.wait(0.5)

			billboardGui:Destroy()
			if ghostMiner and ghostMiner.Parent then
				ghostMiner:Destroy()
			end
		else
			if errorMsg == "Not enough tokens" or errorMsg == "Not enough cash" then
				SoundManager.Error()
				costLabel.Text = errorMsg == "Not enough cash" and "Not enough cash!" or "Not enough tokens!"
				costLabel.TextSize = Camera.ViewportSize.X < 900 and 12 or 15
				costLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
				task.wait(2)
				resetCostDisplay()
				pulseTween:Play()

			elseif errorMsg == "Complete Ages 1, 2 and 3 first" then
				SoundManager.Error()
				costLabel.Text = "Complete Ages 1, 2 and 3 first"
				costLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
				costLabel.TextSize = Camera.ViewportSize.X < 900 and 8 or 11
				-- Prompt Robux purchase immediately
				local remote = RepStorage.Events:FindFirstChild("GetLegendaryMineProductId")
				if remote then
					local ok2, productId = pcall(function() return remote:InvokeServer() end)
					if ok2 and productId and productId ~= 0 then
						game:GetService("MarketplaceService"):PromptProductPurchase(player, productId)
						-- Wait for unlock confirmation then clean up
						local unlockEvent = RepStorage.Events:WaitForChild("LegendaryMineUnlocked", 60)
						if unlockEvent then
							local conn
							conn = unlockEvent.OnClientEvent:Connect(function()
								conn:Disconnect()
								if billboardGui and billboardGui.Parent then billboardGui:Destroy() end
								if ghostMiner and ghostMiner.Parent then ghostMiner:Destroy() end
							end)
						end
					end
				end
				task.wait(2)
				resetCostDisplay()
				pulseTween:Play()

			elseif errorMsg == "Already unlocked" then
				costLabel.Text = "✓ Already unlocked!"
				costLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
				task.wait(2)
				resetCostDisplay()
				pulseTween:Play()

			elseif errorMsg then
				costLabel.Text = "❌ " .. errorMsg
				costLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
				task.wait(2)
				resetCostDisplay()
				pulseTween:Play()

			else
				costLabel.Text = "❌ Cannot unlock!"
				costLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
				task.wait(2)
				resetCostDisplay()
				pulseTween:Play()
			end
		end
	end)
end

for _, hotspot in pairs(hotspotFolder:GetChildren()) do
	task.spawn(function()
		local ok, err = pcall(function()
			setupHotspot(hotspot)
		end)
		if not ok then
			warn("Error setting up", hotspot.Name, ":", err)
		end
	end)
end

local SpeedrunStarted = RepStorage:WaitForChild("Events"):WaitForChild("SpeedrunStarted")
SpeedrunStarted.OnClientEvent:Connect(function()
	task.wait(0.5)
	local newHotspotFolder = player.SetPlot.Value:FindFirstChild("MinerHotspots")
	if not newHotspotFolder then return end
	for _, hotspot in pairs(newHotspotFolder:GetChildren()) do
		task.spawn(function()
			pcall(function() setupHotspot(hotspot) end)
		end)
	end
end)

local RestoredMainPlot = RepStorage:WaitForChild("Events"):WaitForChild("RestoredMainPlot")
RestoredMainPlot.OnClientEvent:Connect(function()
	task.wait(0.5)
	local newHotspotFolder = player.SetPlot.Value:FindFirstChild("MinerHotspots")
	if not newHotspotFolder then return end
	for _, hotspot in pairs(newHotspotFolder:GetChildren()) do
		-- Clean up any ghost miners left over from the speedrun session
		local ghost = hotspot:FindFirstChild("GhostMiner")
		if ghost then ghost:Destroy() end

		task.spawn(function()
			pcall(function() setupHotspot(hotspot) end)
		end)
	end
end)

local AllResearchCompleted = RepStorage:WaitForChild("Events"):WaitForChild("AllResearchCompleted")
AllResearchCompleted.OnClientEvent:Connect(function()
	for hotspotId, refreshFn in pairs(researchLockedResetCallbacks) do
		refreshFn()
		researchLockedResetCallbacks[hotspotId] = nil
	end
end)
