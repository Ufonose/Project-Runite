-- ServerScriptService/Items/OreSmelter.lua
local Players = game:GetService("Players")
local RepStorage = game:GetService("ReplicatedStorage")
local CollectionService = game:GetService("CollectionService")

local ITEM_HANDLER = require(RepStorage.Modules.ItemHandler)
local ITEM_INFO = require(script.Parent.Parent.ItemStats)
local PLOT = script.Parent.Parent.Parent.Parent

local SMELT_INPUT = script.Parent:WaitForChild("SmeltInput")
local SMELT_OUTPUT = script.Parent:WaitForChild("SmeltOutput")
local DISPLAY_SCREEN = script.Parent:WaitForChild("DisplayScreen")

-- Settings
local SPREAD_WIDTH = 2.7
local DETECT_INTERVAL = 0.1 
local REQUIRED_ORES = 5
local SMELT_DELAY = 1

-- Tracking
local oreQueue = {}
local isCurrentlySmelting = false
local processedOres = {} -- For the scanning debounce

-- Smelting Recipes
local ACCEPTED_ORES = {
	["Gold Ore_Cleaned"] = {Color = Color3.fromRGB(255, 215, 0), IngotName = "Gold Ingot"},
	["Iron Ore_Cleaned"] = {Color = Color3.fromRGB(180, 180, 190), IngotName = "Iron Ingot"},
	["Copper Ore_Cleaned"] = {Color = Color3.fromRGB(163, 72, 39), IngotName = "Copper Ingot"},
}

-- Function to update display
local function updateDisplay()
	-- Clear existing dots
	for _, child in pairs(dotsFrame:GetChildren()) do
		if child:IsA("Frame") and child.Name == "Dot" then
			child:Destroy()
		end
	end

	-- Calculate dot size and spacing
	local dotSize = 70
	local spacing = 12
	local totalWidth = (dotSize * REQUIRED_ORES) + (spacing * (REQUIRED_ORES - 1))

	local startX = (dotsFrame.AbsoluteSize.X - totalWidth) / 2

	-- Create dots for each ore in queue
	for i = 1, REQUIRED_ORES do
		local dot = Instance.new("Frame")
		dot.Name = "Dot"
		dot.Size = UDim2.new(0, dotSize, 0, dotSize)
		dot.Position = UDim2.new(0, startX + ((i - 1) * (dotSize + spacing)), 0.5, 0)
		dot.AnchorPoint = Vector2.new(0, 0.5)

		if oreQueue[i] then
			dot.BackgroundColor3 = oreQueue[i].Color
			dot.BackgroundTransparency = 0
		else
			dot.BackgroundColor3 = Color3.fromRGB(50, 50, 55)
			dot.BackgroundTransparency = 0.5
		end

		dot.BorderSizePixel = 0
		dot.Parent = dotsFrame

		local corner = Instance.new("UICorner")
		corner.CornerRadius = UDim.new(1, 0)
		corner.Parent = dot

		-- Add shine effect for filled dots
		if oreQueue[i] then
			local shine = Instance.new("Frame")
			shine.Size = UDim2.new(0.4, 0, 0.4, 0)
			shine.Position = UDim2.new(0.2, 0, 0.2, 0)
			shine.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
			shine.BackgroundTransparency = 0.7
			shine.BorderSizePixel = 0
			shine.Parent = dot

			local shineCorner = Instance.new("UICorner")
			shineCorner.CornerRadius = UDim.new(1, 0)
			shineCorner.Parent = shine
		end
	end

	-- Update status text
	if isCurrentlySmelting then
		statusLabel.Text = "SMELTING..."
		statusLabel.TextColor3 = Color3.fromRGB(255, 150, 50)
	elseif #oreQueue == REQUIRED_ORES then
		statusLabel.Text = "READY TO SMELT"
		statusLabel.TextColor3 = Color3.fromRGB(100, 255, 100)
	else
		statusLabel.Text = #oreQueue .. "/" .. REQUIRED_ORES .. " Ores"
		statusLabel.TextColor3 = Color3.fromRGB(200, 200, 200)
	end
end

local function areAllSameType()
	if #oreQueue == 0 then return false end
	local firstType = oreQueue[1].Type
	for i = 2, #oreQueue do
		if oreQueue[i].Type ~= firstType then return false end
	end
	return true
end

local function createIngot()
	local ingotName = areAllSameType() and ACCEPTED_ORES[oreQueue[1].Type].IngotName or "Impure Ingot"

	local oreFolder = RepStorage:FindFirstChild("OreModels")
	local ingotModel = oreFolder and oreFolder:FindFirstChild(ingotName)
	if not ingotModel then return end

	local newIngot = ingotModel:Clone()
	newIngot.Name = ingotName

	-- Setup Tags for Local Conveyor logic
	CollectionService:AddTag(newIngot, "MovingOre")
	newIngot:SetAttribute("OreID", math.random(1, 1000000))
	newIngot:SetAttribute("LaneOffset", (math.random(-40, 40) / 100))

	-- Custom Tag for Smelting Output
	local ingotTag = Instance.new("StringValue", newIngot)
	ingotTag.Name = "OreModelName" -- Keeping name consistent for potential further processing
	ingotTag.Value = ingotName

	-- Cash calculation
	local totalCash = 0
	for _, ore in ipairs(oreQueue) do totalCash = totalCash + ore.CashValue end

	local cashValue = Instance.new("NumberValue", newIngot)
	cashValue.Name = "Cash"
	cashValue.Value = (ingotName == "Impure Ingot") and 100 or (totalCash * 2)

	-- Spawn and Physics
	local randomX = math.random(-SPREAD_WIDTH/2 * 100, SPREAD_WIDTH/2 * 100) / 100
	newIngot:PivotTo(SMELT_OUTPUT.CFrame * CFrame.new(randomX, 0, 0))
	newIngot.Parent = PLOT.MiscInfo.OresDropped

	local p = (newIngot:IsA("Model") and newIngot.PrimaryPart) or newIngot
	if p:IsA("BasePart") then
		p.Anchored = false
		-- Set Ownership to Plot Owner
		local plotOwner = Players:FindFirstChild(PLOT.Owner.Value)
		if plotOwner then p:SetNetworkOwner(plotOwner) end
	end
end

local function processOre(ORE)
	if isCurrentlySmelting or #oreQueue >= REQUIRED_ORES then
		ORE:Destroy()
		return
	end

	local oreTag = ORE:FindFirstChild("OreModelName")
	local oreType = oreTag and oreTag.Value or ""

	-- DESTROY everything that is NOT Cleaned
	if not string.find(oreType, "_Cleaned") or not ACCEPTED_ORES[oreType] then
		ORE:Destroy()
		return
	end

	-- Add to queue
	local cash = ORE:FindFirstChild("Cash") and ORE.Cash.Value or 0
	table.insert(oreQueue, {
		Type = oreType,
		Color = ACCEPTED_ORES[oreType].Color,
		CashValue = cash
	})

	ORE:Destroy()
	updateDisplay()

	if #oreQueue >= REQUIRED_ORES then
		isCurrentlySmelting = true
		updateDisplay()
		task.wait(SMELT_DELAY)
		createIngot()
		oreQueue = {}
		isCurrentlySmelting = false
		updateDisplay()
	end
end

-- ============================================================
-- THE DETECTION LOOP (Scanning Mode)
-- ============================================================
task.spawn(function()
	local overlapParams = OverlapParams.new()
	overlapParams.FilterType = Enum.RaycastFilterType.Include

	while true do
		task.wait(DETECT_INTERVAL)
		overlapParams.FilterDescendantsInstances = CollectionService:GetTagged("MovingOre")

		local partsInInput = workspace:GetPartBoundsInBox(SMELT_INPUT.CFrame, SMELT_INPUT.Size, overlapParams)

		for _, hit in pairs(partsInInput) do
			local ORE = nil
			if hit:FindFirstChild("Cash") then
				ORE = hit
			elseif hit.Parent and hit.Parent:FindFirstChild("Cash") then
				ORE = hit.Parent
			end

			if ORE and not processedOres[ORE] then
				processedOres[ORE] = true

				-- Use ItemStats to upgrade before consuming the ore
				ITEM_HANDLER.UPGRADE_ORE(PLOT, script.Parent.Parent, ORE, ITEM_INFO.Addition, ITEM_INFO.Multiplier, ITEM_INFO.ItemId, 30)

				processOre(ORE)

				task.delay(0.5, function() processedOres[ORE] = nil end)
			end
		end
	end
end)

updateDisplay()
print("Blast Furnace Scanning Logic Loaded!")
