local ITEM_HANDLER = require(RepStorage.Modules.ItemHandler)

-- Settings
local SPREAD_WIDTH = 2.7
local DETECT_INTERVAL = 0.1

local SMELT_DELAY = 1

-- Tracking
local oreQueue = {}
local isCurrentlySmelting = false
local processedOres = {} 

-- Smelting Recipes
local ACCEPTED_ORES = {
	["Gold Ore_Cleaned"] = {Color = Color3.fromRGB(255, 215, 0), IngotName = "Gold Ingot"},
	["Iron Ore_Cleaned"] = {Color = Color3.fromRGB(180, 180, 190), IngotName = "Iron Ingot"},
	["Copper Ore_Cleaned"] = {Color = Color3.fromRGB(163, 72, 39), IngotName = "Copper Ingot"},
}

-- ── DISPLAY & PROCESSING FUNCTIONS ───────────────────────────────────────

local function updateDisplay()
	-- Clear existing dots
	for _, child in pairs(dotsFrame:GetChildren()) do
		if child:IsA("Frame") and child.Name == "Dot" then child:Destroy() end
	end

	local dotSize, spacing = 70, 12
	local totalWidth = (dotSize * REQUIRED_ORES) + (spacing * (REQUIRED_ORES - 1))
	local startX = (dotsFrame.AbsoluteSize.X - totalWidth) / 2

	for i = 1, REQUIRED_ORES do
		local dot = Instance.new("Frame")
		dot.Name = "Dot"; dot.Size = UDim2.new(0, dotSize, 0, dotSize)
		dot.Position = UDim2.new(0, startX + ((i - 1) * (dotSize + spacing)), 0.5, 0)
		dot.AnchorPoint = Vector2.new(0, 0.5)
		dot.BackgroundColor3 = oreQueue[i] and oreQueue[i].Color or Color3.fromRGB(50, 50, 55)
		dot.BackgroundTransparency = oreQueue[i] and 0 or 0.5
		dot.Parent = dotsFrame
		local corner = Instance.new("UICorner"); corner.CornerRadius = UDim.new(1, 0); corner.Parent = dot
	end

	if isCurrentlySmelting then
		statusLabel.Text = "SMELTING..."
		statusLabel.TextColor3 = Color3.fromRGB(255, 150, 50)
	elseif #oreQueue == REQUIRED_ORES then
		statusLabel.Text = "READY TO SMELT"
		statusLabel.TextColor3 = Color3.fromRGB(100, 255, 100)
	else
		statusLabel.Text = #oreQueue .. "/" .. REQUIRED_ORES .. " Ores"
		statusLabel.TextColor3 = Color3.fromRGB(200, 200, 200)
	end
end

local function createIngot()
	local allSame = true
	local firstType = oreQueue[1].Type
	for i = 2, #oreQueue do if oreQueue[i].Type ~= firstType then allSame = false break end end

	local ingotName = allSame and ACCEPTED_ORES[firstType].IngotName or "Impure Ingot"
	local ingotModel = RepStorage.OreModels:FindFirstChild(ingotName)
	if not ingotModel then return end

	local newIngot = ingotModel:Clone()
	newIngot.Name = ingotName
	CollectionService:AddTag(newIngot, "MovingOre")
	newIngot:SetAttribute("OreID", math.random(1, 1000000))

	local ingotTag = Instance.new("StringValue", newIngot)
	ingotTag.Name = "OreModelName"; ingotTag.Value = ingotName

	local totalCash = 0
	for _, ore in ipairs(oreQueue) do totalCash = totalCash + ore.CashValue end
	local cashValue = Instance.new("NumberValue", newIngot)
	cashValue.Name = "Cash"
	cashValue.Value = (ingotName == "Impure Ingot") and 100 or (totalCash * 2)

	local randomX = math.random(-SPREAD_WIDTH/2 * 100, SPREAD_WIDTH/2 * 100) / 100
	newIngot:PivotTo(SMELT_OUTPUT.CFrame * CFrame.new(randomX, 0, 0))
	newIngot.Parent = PLOT.MiscInfo.OresDropped

	local p = (newIngot:IsA("Model") and newIngot.PrimaryPart) or newIngot
	if p:IsA("BasePart") then
		p.Anchored = false
		local plotOwner = Players:FindFirstChild(PLOT.Owner.Value)
		if plotOwner then p:SetNetworkOwner(plotOwner) end
	end
end

-- ── ORE PROCESSING & DETECTION ───────────────────────────────────────────

local function processOre(ORE)
	if isCurrentlySmelting or #oreQueue >= REQUIRED_ORES then
		ORE:Destroy()
		return
	end

	local oreTag = ORE:FindFirstChild("OreModelName")
	local oreType = oreTag and oreTag.Value or ""

	if not string.find(oreType, "_Cleaned") or not ACCEPTED_ORES[oreType] then
		ORE:Destroy()
		return
	end

	table.insert(oreQueue, {
		Type = oreType,
		Color = ACCEPTED_ORES[oreType].Color,
		CashValue = ORE:FindFirstChild("Cash") and ORE.Cash.Value or 0
	})

	ORE:Destroy()
	updateDisplay()

	if #oreQueue >= REQUIRED_ORES then
		isCurrentlySmelting = true
		updateDisplay() -- GUI changes to "SMELTING..." here

		-- TRIGGER CRUSH SOUND ON SMELT START
		if CRUSH_SOUND then
			CRUSH_SOUND:Stop()
			CRUSH_SOUND.PlaybackSpeed = math.random(70, 95) / 100 -- Deeper pitch for furnace
			CRUSH_SOUND:Play()
		end

		task.wait(SMELT_DELAY)
		createIngot()
		oreQueue = {}
		isCurrentlySmelting = false
		updateDisplay()
	end
end

task.spawn(function()
	local overlapParams = OverlapParams.new()
	overlapParams.FilterType = Enum.RaycastFilterType.Include

	while true do
		task.wait(DETECT_INTERVAL)
		overlapParams.FilterDescendantsInstances = CollectionService:GetTagged("MovingOre")

		local partsInInput = workspace:GetPartBoundsInBox(SMELT_INPUT.CFrame, SMELT_INPUT.Size, overlapParams)

		for _, hit in pairs(partsInInput) do
			local ORE = hit:FindFirstChild("Cash") and hit or (hit.Parent and hit.Parent:FindFirstChild("Cash") and hit.Parent)

			if ORE and not processedOres[ORE] then
				processedOres[ORE] = true
				ITEM_HANDLER.UPGRADE_ORE(PLOT, script.Parent.Parent, ORE, ITEM_INFO.Addition, ITEM_INFO.Multiplier, ITEM_INFO.ItemId, 30)
				processOre(ORE)
				task.delay(0.5, function() processedOres[ORE] = nil end)
			end
		end
	end
end)

updateDisplay()
print("Blast Furnace Scanning Logic Loaded with Audio Trigger!")
