-- Services
local RepStorage = game:GetService("ReplicatedStorage")
local ServerStorage = game:GetService("ServerStorage")
local Plrs = game:GetService("Players")

-- Modules
local MAIN_LIB = require(RepStorage.Modules.MainLib)
local PlayerSession = require(ServerStorage.Modules.PlayerDataManager)

-- Variables
local DEBOUNCE = {}
local MINIMUM_ITEM_QUANTITIES = {
	["Basic Conveyor"] = 8,
	["Corner Conveyor"] = 2,
}

function PLACE_ITEM(Plr, PLACED_ITEMS, ITEMS)
	if not (Plr and PLACED_ITEMS and ITEMS) then return false end
	local session = PlayerSession.Get(Plr)
	if not session then return false end

	local ITEM_FAILED = false
	for I, ITEM in pairs(ITEMS) do
		local ITEM_REP = RepStorage.Items:FindFirstChild(ITEM["ITEM_NAME"])
		if not ITEM_REP then continue end

		local ITEM_INFO = require(ITEM_REP.ItemStats)
		local itemKey = tostring(ITEM_INFO.ItemId)

		-- Check Inventory via Module
		if session["Items"][itemKey] and session["Items"][itemKey]["Amount"] > 0 then
			local REAL_ITEM = ITEM_REP:Clone()
			REAL_ITEM:SetPrimaryPartCFrame(ITEM["ITEM_LOCATION"])

			-- Handle Placement Layers (From your DataHandling logic)
			local layerValue = Instance.new("IntValue", REAL_ITEM)
			layerValue.Name = "PlacementLayer"
			if type(ITEM_INFO.PlacementLayer) == "table" then
				layerValue.Value = ITEM_INFO.PlacementLayer[1]
				local layersString = Instance.new("StringValue", REAL_ITEM)
				layersString.Name = "PlacementLayers"
				layersString.Value = table.concat(ITEM_INFO.PlacementLayer, ",")
			else
				layerValue.Value = ITEM_INFO.PlacementLayer or 1
			end

			-- Handle Conveyor Reversal Attributes
			if ITEM["CONVEYOR_REVERSED"] then
				REAL_ITEM:SetAttribute("ConveyorReversed", true)
				for _, descendant in pairs(REAL_ITEM:GetDescendants()) do
					if descendant:IsA("BasePart") and (descendant.Name == "Conv" or descendant.Name == "ConvSlope" or descendant.Name == "SideBeam" or descendant.Name == "SideStuck") then
						if not descendant:GetAttribute("NoFlip") and descendant.Name ~= "ConvMerge" then
							local cv = descendant:FindFirstChild("ConvVelo")
							if cv then cv.Value = -cv.Value end
							local beam = descendant:FindFirstChildOfClass("Beam")
							if beam then beam.TextureSpeed = -beam.TextureSpeed end
						end
					end
				end
			end

			-- Handle Sorter/Assembly Configs
			if ITEM["SORT_CONFIG"] then
				local sortConfig = REAL_ITEM:FindFirstChild("SortConfig") or Instance.new("Folder", REAL_ITEM)
				sortConfig.Name = "SortConfig"
				for portId, oreType in pairs(ITEM["SORT_CONFIG"]) do
					local portValue = sortConfig:FindFirstChild(portId) or Instance.new("StringValue", sortConfig)
					portValue.Name = portId
					portValue.Value = oreType
				end
			end

			REAL_ITEM.Parent = PLACED_ITEMS

			-- Apply Global Upgrades (Conveyors/Miners)
			pcall(function()
				if _G.ApplyConveyorUpgradesToItem then _G.ApplyConveyorUpgradesToItem(Plr, REAL_ITEM) end
				if _G.ApplyMinerUpgradesToItem then _G.ApplyMinerUpgradesToItem(Plr, REAL_ITEM) end
			end)

			-- Deduct Inventory
			session["Items"][itemKey]["Amount"] -= 1

			task.spawn(function()
				for _, D in pairs(REAL_ITEM:GetDescendants()) do
					if D:IsA("Script") then D.Disabled = D:FindFirstChild("ForceDisabled") end
				end
				RepStorage.Events.InventoryUpdate:FireClient(Plr)
			end)
		else
			ITEM_FAILED = true
		end
	end
	return not ITEM_FAILED
end

function GET_INVENTORY(Plr, INVENTORY_TYPE)
	local session = PlayerSession.Get(Plr)
	return session and session[INVENTORY_TYPE] or {}
end

function BUY_ITEM(Plr, ITEM_ID, AMOUNT)
	local CASH = ServerStorage.ServerLockedPlayerInfo.PlayerCash:FindFirstChild(Plr.Name)
	if not CASH then return false end

	local CANT_BUY_TYPES = {"special", "cantbuy"}
	local amt = AMOUNT or 1
	local IDs = type(ITEM_ID) == "table" and ITEM_ID or {ITEM_ID}
	local costTotal = 0
	local validIDs = {}

	for _, id in pairs(IDs) do
		local itemModel = MAIN_LIB.ITEM_FROM_ID(id)
		local stats = require(itemModel.ItemStats)
		if not table.find(CANT_BUY_TYPES, stats.ItemType) then
			if CASH.Value >= math.floor(stats.Cost * amt) then
				costTotal += math.floor(stats.Cost * amt)
				table.insert(validIDs, id)
			end
		end
	end

	if #validIDs == #IDs then
		CASH.Value -= costTotal
		for _, id in pairs(validIDs) do
			ItemGive(Plr, id, amt)
		end
		return true
	end
	return false
end

function SELL_ITEM(Plr, ITEMS)
	local CASH = ServerStorage.ServerLockedPlayerInfo.PlayerCash:FindFirstChild(Plr.Name)
	local session = PlayerSession.Get(Plr) -- Get ProfileStore data

	if CASH and session then
		local CANT_SELL_TYPES = {"special", "cantbuy"}
		local ITEM_LIST = type(ITEMS) == "table" and ITEMS or {ITEMS}

		-- 1. Count items being sold by name
		local itemsBeingSold = {}
		for _, item in pairs(ITEM_LIST) do
			local stats = require(item.ItemStats)
			local itemName = MAIN_LIB.ITEM_FROM_ID(stats.ItemId).Name
			itemsBeingSold[itemName] = (itemsBeingSold[itemName] or 0) + 1
		end

		-- 2. Count TOTAL items (placed on plot + in inventory)
		-- Adjust this path if your plot system uses a different folder name
		local PLR_PLOT = Plr:FindFirstChild("SetPlot") and Plr.SetPlot.Value
		local PLACED_ITEMS = PLR_PLOT and PLR_PLOT:FindFirstChild("PlacedItems")

		local totalItemCounts = {}

		-- Count items currently placed on the plot
		if PLACED_ITEMS then
			for _, item in pairs(PLACED_ITEMS:GetChildren()) do
				if item:FindFirstChild("ItemStats") then
					local itemStats = require(item.ItemStats)
					local itemName = MAIN_LIB.ITEM_FROM_ID(itemStats.ItemId).Name
					totalItemCounts[itemName] = (totalItemCounts[itemName] or 0) + 1
				end
			end
		end

		-- Count items currently in ProfileStore inventory
		for itemId, itemData in pairs(session["Items"]) do
			if itemData.Amount and itemData.Amount > 0 then
				local itemTemplate = MAIN_LIB.ITEM_FROM_ID(tonumber(itemId))
				if itemTemplate then
					local itemName = itemTemplate.Name
					totalItemCounts[itemName] = (totalItemCounts[itemName] or 0) + itemData.Amount
				end
			end
		end

		-- 3. Check if selling would drop below minimum
		for itemName, sellCount in pairs(itemsBeingSold) do
			local minimumRequired = MINIMUM_ITEM_QUANTITIES[itemName]
			if minimumRequired then
				local totalOwned = totalItemCounts[itemName] or 0
				local remainingAfterSale = totalOwned - sellCount

				if remainingAfterSale < minimumRequired then
					RepStorage.Events.MessageClient:FireClient(Plr, "screen", {
						["message"] = "Cannot sell! You must keep at least " .. minimumRequired .. " " .. itemName .. "s."
					})
					return false -- Stop the whole sale
				end
			end
		end

		-- 4. Proceed with selling if checks pass
		for _, item in pairs(ITEM_LIST) do
			-- Check hotspot attributes
			if item:GetAttribute("HotspotMiner") or item:GetAttribute("HotspotFurnace") then
				RepStorage.Events.MessageClient:FireClient(Plr, "screen", {["message"] = "Hotspot items cannot be sold!"})
				continue 
			end

			local ITEM_INFO = require(item.ItemStats)
			if not table.find(CANT_SELL_TYPES, ITEM_INFO.ItemType) then
				local sellValue = math.floor(ITEM_INFO.Cost * 0.3)
				CASH.Value += sellValue
				item:Destroy()
			end
		end
		return true
	end
	return false
end

function WITHDRAW_ITEMS(Plr, ITEMS)
	local session = PlayerSession.Get(Plr)
	if not session or DEBOUNCE[Plr.Name] then return false end

	DEBOUNCE[Plr.Name] = true
	task.delay(0.3, function() DEBOUNCE[Plr.Name] = nil end)

	local ITEM_LIST = type(ITEMS) == "table" and ITEMS or {ITEMS}
	for _, item in pairs(ITEM_LIST) do
		if item:GetAttribute("HotspotMiner") or item:GetAttribute("HotspotFurnace") then continue end

		local stats = require(item.ItemStats)
		session["Items"][tostring(stats.ItemId)]["Amount"] += 1
		item:Destroy()
	end
	RepStorage.Events.InventoryUpdate:FireClient(Plr)
	return true
end

function ItemGive(Plr, ITEM_ID, AMOUNT)
	local session = PlayerSession.Get(Plr)
	local idStr = tostring(ITEM_ID)
	if session and session["Items"][idStr] then
		session["Items"][idStr]["Amount"] += AMOUNT
		RepStorage.Events.MessageClient:FireClient(Plr, "screen", {["message"] = AMOUNT.." "..MAIN_LIB.ITEM_FROM_ID(ITEM_ID).Name.."s added to your Inventory"})
		RepStorage.Events.InventoryUpdate:FireClient(Plr)
		return true
	end
	return false
end

function toggle_update(plr, toggle_name)
	local setting = plr.Settings:FindFirstChild(toggle_name)
	if setting then setting.Value = not setting.Value return true end
	return false
end

-- Remote Listeners
RepStorage.Events.GetInventoryType.OnServerInvoke = GET_INVENTORY
RepStorage.Events.WithdrawItems.OnServerInvoke = WITHDRAW_ITEMS
RepStorage.Events.UpdateToggle.OnServerInvoke = toggle_update
RepStorage.Events.PlaceItem.OnServerInvoke = PLACE_ITEM
RepStorage.Events.SellItem.OnServerInvoke = SELL_ITEM
RepStorage.Events.BuyItem.OnServerInvoke = BUY_ITEM
ServerStorage.Events.ItemGive.OnInvoke = ItemGive
