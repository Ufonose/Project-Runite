-- Services
local RepStorage = game:GetService("ReplicatedStorage")
local ServerStorage = game:GetService("ServerStorage")
local Plrs = game:GetService("Players")
local HttpService = game:GetService("HttpService")

-- Modules
local DATA_FUNCTION_HANDLING = require(ServerStorage.Modules.DataHandling)
local PLAYER_DATA_INFO = require(ServerStorage.Modules.PlayerDataInfo)
local CASH_LIB = require(RepStorage.Modules.CashLib)
local MainLib = require(RepStorage.Modules.MainLib)

-- OUR NEW MODULE (ProfileStore Powered)
local PlayerSession = require(ServerStorage.Modules.PlayerDataManager)

-- This function is now just for PREPARING data before a session ends
local function PREPARE_FINAL_SAVE(Plr)
	local session = PlayerSession.Get(Plr)
	if not session then return end

	-- 1. Save Basic Values into Session
	for val_name, _ in pairs(PLAYER_DATA_INFO.SIMPLE_VALS) do
		local val_obj = Plr:FindFirstChild(val_name)
		if val_obj then session[val_name] = val_obj.Value end
	end

	-- 2. Save Settings into Session
	for val_name, _ in pairs(PLAYER_DATA_INFO.SETTINGS_VALUES) do
		if val_name == "MinesActive" then continue end
		local val_obj = Plr.Settings:FindFirstChild(val_name)
		if val_obj then session["setting"..val_name] = val_obj.Value end
	end

	-- 3. Save Cash into Session
	local plr_cash = ServerStorage.ServerLockedPlayerInfo.PlayerCash:FindFirstChild(Plr.Name)
	if plr_cash then session["Money"] = plr_cash.Value end

	-- 4. Save Plot
	session["PlotItemPlaced"] = DATA_FUNCTION_HANDLING.UNLOAD_PLOT(Plr)
end

local function LOAD_DATA(Plr)
	if Plr:FindFirstChild("Loading") then return end
	local loadingTag = Instance.new("BoolValue", Plr)
	loadingTag.Name = "Loading"

	-- 1. Start ProfileStore Session
	local session = PlayerSession.Load(Plr) 
	if not session then return end

	-- 2. Setup Simple Values (Rebirths, etc)
	for val_name, val_data in pairs(PLAYER_DATA_INFO.SIMPLE_VALS) do
		local v = Instance.new(val_data["VALUE_TYPE"], Plr)
		v.Name = val_name
		v.Value = session[val_name] or val_data["DEFAULT"]
		v.Changed:Connect(function(newVal) session[val_name] = newVal end)
	end

	-- 3. Setup Settings
	local settingsFolder = Instance.new("Folder", Plr)
	settingsFolder.Name = "Settings"

	for val_name, val_data in pairs(PLAYER_DATA_INFO.SETTINGS_VALUES) do
		local v = Instance.new(val_data["VALUE_TYPE"] or "BoolValue", settingsFolder)
		v.Name = val_name

		-- FIX: Check if the value exists in the session first (even if it's false)
		local savedValue = session["setting"..val_name]

		if val_name == "MinesActive" then
			v.Value = true
		elseif savedValue ~= nil then
			v.Value = savedValue -- This correctly picks up "false" (Off)
		else
			v.Value = val_data["DEFAULT"]
		end

		-- Sync changes immediately
		v.Changed:Connect(function(newVal) 
			session["setting"..val_name] = newVal 
		end)
	end

	-- 4. Internal Cash Logic
	local plr_cash = Instance.new("NumberValue", ServerStorage.ServerLockedPlayerInfo.PlayerCash)
	plr_cash.Name = Plr.Name
	plr_cash.Value = session["Money"] or 400

	-- 5. Inventory Setup
	local default_inv = {["2"] = {["Amount"] = 18, ["Unlocked"] = true}, ["7"] = {["Amount"] = 4, ["Unlocked"] = true}}
	if not session["Items"] or next(session["Items"]) == nil then
		session["Items"] = default_inv
	end

	for _, item in pairs(RepStorage.Items:GetChildren()) do
		local stats = item:FindFirstChild("ItemStats")
		if stats then
			local id = tostring(require(stats).ItemId)
			if not session["Items"][id] then
				session["Items"][id] = {["Unlocked"] = false, ["Amount"] = 0}
			end
		end
	end

	-- 6. Plot Loading
	local plot_table = session["PlotItemPlaced"] or {}
	DATA_FUNCTION_HANDLING.LOAD_PLOT(Plr, plot_table)

	-- 7. Leaderstats Setup (Visualizing Money as Cash)
	local ls = Instance.new("Folder", Plr); ls.Name = "leaderstats"
	local cStat = Instance.new("StringValue", ls); cStat.Name = "Cash"
	local rStat = Instance.new("StringValue", ls); rStat.Name = "Rebirths"

	local function updateVisuals()
		session["Money"] = plr_cash.Value 
		cStat.Value = "$" .. CASH_LIB.SUFFIX_NUM(plr_cash.Value)
		local rb = Plr:FindFirstChild("Rebirth")
		rStat.Value = tostring(rb and (rb.Value + 1) or 1)
	end

	plr_cash.Changed:Connect(updateVisuals)
	if Plr:FindFirstChild("Rebirth") then Plr.Rebirth.Changed:Connect(updateVisuals) end
	updateVisuals()

	-- 8. Final Completion Signal
	local dsTag = Instance.new("BoolValue")
	dsTag.Name = "DataLoaded"
	dsTag.Parent = Plr 

	loadingTag:Destroy()
	print("✅ LOAD_DATA finalized for " .. Plr.Name)
end

-- Events
Plrs.PlayerRemoving:Connect(function(Plr)
	PREPARE_FINAL_SAVE(Plr)
	MainLib.CLEAR_PLOT(Plr)

	local c = ServerStorage.ServerLockedPlayerInfo.PlayerCash:FindFirstChild(Plr.Name)
	if c then c:Destroy() end

	PlayerSession.Remove(Plr) -- This calls EndSession() which saves to DB
end)

Plrs.PlayerAdded:Connect(LOAD_DATA)

game:BindToClose(function()
	for _, plr in pairs(Plrs:GetPlayers()) do
		task.spawn(function() 
			PREPARE_FINAL_SAVE(plr)
			PlayerSession.Remove(plr)  -- EndSession blocks until save completes
		end)
	end
end)
