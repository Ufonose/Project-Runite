-- Services
local RepStorage = game:GetService("ReplicatedStorage")
local ServerStorage = game:GetService("ServerStorage")
local Plrs = game:GetService("Players")

-- Modules
local MAIN_LIB = require(RepStorage.Modules.MainLib)
local CASH_LIB = require(RepStorage.Modules.CashLib)
local PlayerSession = require(ServerStorage.Modules.PlayerDataManager) -- ADDED THIS

local function SORT_ITEMS(Plr)
	local AVAILABLE_ITEMS = {}
	for I, ITEM in pairs(RepStorage.Items:GetChildren()) do
		local ITEM_INFO = require(ITEM.ItemStats)
		-- Checks if item is a reborn item
		if ITEM_INFO.SpecialTags["RebirthChance"] and ITEM_INFO.SpecialTags["RebirthReq"] then
			-- Checks if player's rebirth level is high enough
			if Plr.Rebirth.Value >= ITEM_INFO.SpecialTags.RebirthReq then
				AVAILABLE_ITEMS[#AVAILABLE_ITEMS+1] = ITEM
			end
		end
	end
	return AVAILABLE_ITEMS
end

function IS_REBIRTH_PROOF(ITEM)
	local ITEM_INFO = require(ITEM.ItemStats)
	local TIER_NAME, TIER_DATA = MAIN_LIB.GET_ITEM_TIER_INFO(ITEM.Name)
	local IS_PROOF = false

	if ITEM_INFO.SpecialTags["RebornProof"] then
		IS_PROOF = true
	end

	if TIER_DATA and TIER_DATA["RebornProof"] then
		IS_PROOF = true
	end
	return IS_PROOF
end

function CLEAR_PLOT(Plr, PLR_PLOT)
	if Plr then
		local session = PlayerSession.Get(Plr)
		if not session then return end

		-- 1. Clear items from the physical plot
		for i, ITEM in pairs(PLR_PLOT.PlacedItems:GetChildren()) do
			local ITEM_INFO = require(ITEM.ItemStats)
			if IS_REBIRTH_PROOF(ITEM) then
				-- Give player items back from plot to MODULE inventory if rebirth proof
				local idStr = tostring(ITEM_INFO.ItemId)
				if session["Items"][idStr] then
					session["Items"][idStr]["Amount"] += 1
				end
			end
			ITEM:Destroy()
		end

		-- 2. Wipe non-rebirth proof items from MODULE inventory
		for ID, DATA_ITEM in pairs(session["Items"]) do
			if DATA_ITEM and DATA_ITEM["Amount"] ~= 0 then
				local ITEM_MODEL = MAIN_LIB.ITEM_FROM_ID(tonumber(ID))
				if ITEM_MODEL and not IS_REBIRTH_PROOF(ITEM_MODEL) then
					DATA_ITEM["Amount"] = 0 -- Removes item from inventory
				end
			end
		end

		-- Notify client that inventory changed
		RepStorage.Events.InventoryUpdate:FireClient(Plr)
	end
end

function REBIRTH(Plr)
	if Plr then
		local session = PlayerSession.Get(Plr)
		local PLR_PLOT = Plr.SetPlot.Value
		local plrCash = ServerStorage.ServerLockedPlayerInfo.PlayerCash:FindFirstChild(Plr.Name)

		if not session or not plrCash or not PLR_PLOT then return false end

		if plrCash.Value >= CASH_LIB.GET_REBIRTH_COST(Plr) then
			local ITEMS_OBTAINABLE = SORT_ITEMS(Plr)

			if #ITEMS_OBTAINABLE < 2 then
				RepStorage.Events.MessageClient:FireClient(Plr, "screen", {["message"] = "REBIRTH FAILED: Not enough Rebirth Items exist in game!"})
				warn("At least 2 rebirth items with RebirthReq 0 are needed for rebirthing to work!")
				return false
			end

			-- Increase Rebirth Value
			Plr.Rebirth.Value += 1

			-- Clear Ores and Plot
			PLR_PLOT.MiscInfo.OresDropped:ClearAllChildren()
			CLEAR_PLOT(Plr, PLR_PLOT)

			-- Reset Cash
			plrCash.Value = 100

			-- Give Starter Items (Module-based)
			if session["Items"]["2"] then session["Items"]["2"]["Amount"] = 18 end -- Conveyors
			if session["Items"]["7"] then session["Items"]["7"]["Amount"] = 4 end  -- Corners
			-- Add any other starters here (e.g. Stone Mine)

			-- Choose and give reward
			local CHOSEN_ITEM = ITEMS_OBTAINABLE[math.random(1, #ITEMS_OBTAINABLE)]
			local CHOSEN_ITEM_INFO = require(CHOSEN_ITEM.ItemStats)

			-- Use the Invoke to give the item properly
			ServerStorage.Events.ItemGive:Invoke(Plr, CHOSEN_ITEM_INFO.ItemId, 1)

			return true
		end
	end
	return false
end

RepStorage.Events.Rebirth.OnServerInvoke = REBIRTH
