-- ServerScriptService/PlotHandling

-- Services
local RepStorage = game:GetService("ReplicatedStorage")
local ServerStorage = game:GetService("ServerStorage")
local Plrs = game:GetService("Players")

-- Modules
local Enums = require(RepStorage.Modules.CustomEnums)
local MainLib = require(RepStorage.Modules.MainLib)

-- ============================================================
-- TEMPLATE STORAGE SETUP
-- Store a clean copy of each plot's CaveUnlocks and MinerHotspots
-- BEFORE any player can modify them, so we can restore on leave
-- ============================================================
local templateStorage = ServerStorage:FindFirstChild("PlotTemplates")
if not templateStorage then
	templateStorage = Instance.new("Folder")
	templateStorage.Name = "PlotTemplates"
	templateStorage.Parent = ServerStorage
end

for I, PLOT in pairs(workspace.Plots:GetChildren()) do
	PLOT.Owner.Value = "Empty Plot"

	-- Save clean template for this plot
	local plotTemplate = Instance.new("Folder")
	plotTemplate.Name = PLOT.Name
	plotTemplate.Parent = templateStorage

	local caveUnlocks = PLOT:FindFirstChild("CaveUnlocks")
	if caveUnlocks then
		caveUnlocks:Clone().Parent = plotTemplate
	end

	local minerHotspots = PLOT:FindFirstChild("MinerHotspots")
	if minerHotspots then
		minerHotspots:Clone().Parent = plotTemplate
	end

	-- Misc folders setup
	for I, MISC_FOLDER_NAME in pairs(Enums.Arrays.MISC_PLOT_FOLDERS) do
		local FOLDER = Instance.new("Folder", PLOT.MiscInfo)
		FOLDER.Name = MISC_FOLDER_NAME
	end

	-- Destroy ores that fall onto the BasePlot
	PLOT.BasePlot.Touched:Connect(function(HIT)
		local ORE_TO_DESTROY = nil

		if HIT:FindFirstChild("Cash") then
			ORE_TO_DESTROY = HIT
		elseif HIT.Parent and HIT.Parent:IsA("Model") and HIT.Parent:FindFirstChild("Cash") then
			ORE_TO_DESTROY = HIT.Parent
		end

		if ORE_TO_DESTROY then
			ORE_TO_DESTROY:Destroy()
		end
	end)
end

-- ============================================================
-- PLAYER FUNCTIONS
-- ============================================================

local function SPAWN_PLAYER(Plr)
	Plr.Character:WaitForChild("HumanoidRootPart")
	local REAL_PLOT = Plr.SetPlot.Value

	local spawnPart = REAL_PLOT:FindFirstChild("SpawnLocation")

	if spawnPart then
		for I = 1, 5 do
			Plr.Character:WaitForChild("HumanoidRootPart")
			Plr.Character.HumanoidRootPart.CFrame = CFrame.new(spawnPart.Position + Vector3.new(0, Enums.Nums.SPAWN_REG_HEIGHT, 0))
			wait()
		end
	else
		warn("SpawnLocation not found in plot for player:", Plr.Name)
	end
end

local function PLR_ADDED(Plr)
	local PLOT_INFO = Instance.new("ObjectValue", Plr)
	PLOT_INFO.Name = "SetPlot"
	PLOT_INFO.Value = MainLib.SET_PLOT(Plr)

	PLOT_INFO.Value.Owner.Value = Plr.Name

	Plr.CharacterAdded:Connect(function()
		SPAWN_PLAYER(Plr)
	end)
end

Plrs.PlayerAdded:Connect(PLR_ADDED)
