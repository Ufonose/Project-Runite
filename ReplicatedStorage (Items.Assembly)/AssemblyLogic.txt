-- ServerScriptService/Items/PlateProcessor.lua
local Players = game:GetService("Players")
local RepStorage = game:GetService("ReplicatedStorage")
local CollectionService = game:GetService("CollectionService")

local ITEM_HANDLER = require(RepStorage.Modules.ItemHandler)
local ITEM_INFO = require(script.Parent.Parent.ItemStats)
local PLOT = script.Parent.Parent.Parent.Parent

local MODEL = script.Parent
local ASSEMBLY_INPUT = MODEL:WaitForChild("AssemblyInput")
local ASSEMBLY_OUTPUT = MODEL:WaitForChild("AssemblyOutput")
local DISPLAY_SCREEN = MODEL:WaitForChild("DisplayScreen")
local DISPLAY_SCREEN_2 = MODEL:FindFirstChild("DisplayScreen2") 

-- ── SOUND SETUP ──────────────────────────────────────────────────────────
local CRUSH_SOUND = ASSEMBLY_INPUT:FindFirstChild("Crush")
local SOUND_COOLDOWN = 0.5
local lastSoundTime = 0

-- Config setup
if not script.Parent.Parent:FindFirstChild("AssemblyConfig") then
	local config = Instance.new("StringValue", script.Parent.Parent)
	config.Name = "AssemblyConfig"; config.Value = "Gear"
end
local ASSEMBLY_CONFIG = script.Parent.Parent.AssemblyConfig

local SPREAD_WIDTH, DETECT_INTERVAL = 2.7, 0.1 
local processedItems = {}

-- ── GUI LOGIC ─────────────────────────────────────────────────────────────
local function createDisplayOnScreen(screen)
	if not screen then return end
	local displayGui = screen:FindFirstChild("DisplayGui") or Instance.new("SurfaceGui", screen)
	displayGui.Name = "DisplayGui"; displayGui.Face = Enum.NormalId.Left; displayGui.AlwaysOnTop = false
	displayGui:ClearAllChildren()

	local mainFrame = Instance.new("Frame", displayGui)
	mainFrame.Size = UDim2.new(1, 0, 1, 0); mainFrame.BackgroundColor3 = Color3.fromRGB(20, 20, 25); mainFrame.BorderSizePixel = 0

	local title = Instance.new("TextLabel", mainFrame)
	title.Size = UDim2.new(1, 0, 0.3, 0); title.BackgroundTransparency = 1; title.TextColor3 = Color3.fromRGB(207, 207, 207)
	title.Text = "COMPONENT ASSEMBLER"; title.TextSize = 60; title.Font = Enum.Font.GothamBold

	local outputLabel = Instance.new("TextLabel", mainFrame)
	outputLabel.Size = UDim2.new(1, -20, 0.35, 0); outputLabel.Position = UDim2.new(0, 10, 0.35, 0)
	outputLabel.BackgroundColor3 = Color3.fromRGB(30, 30, 35); outputLabel.TextColor3 = Color3.fromRGB(100, 255, 100)
	outputLabel.Text = "Making: " .. ASSEMBLY_CONFIG.Value; outputLabel.TextSize = 50; outputLabel.Font = Enum.Font.GothamBold
	Instance.new("UICorner", outputLabel).CornerRadius = UDim.new(0, 12)
end

local function updateDisplay()
	createDisplayOnScreen(DISPLAY_SCREEN)
	if DISPLAY_SCREEN_2 then createDisplayOnScreen(DISPLAY_SCREEN_2) end
end
-- ── ASSEMBLY LOGIC ────────────────────────────────────────────────────────
local function assembleItem(ITEM)
	if not ITEM or processedItems[ITEM] then return false end

	local itemModelTag = ITEM:FindFirstChild("OreModelName")
	if not itemModelTag or not string.find(itemModelTag.Value, "Plate") then return false end

	processedItems[ITEM] = true
	local metalType = string.gsub(itemModelTag.Value, " Plate", "")
	local outputModelName = metalType .. " " .. ASSEMBLY_CONFIG.Value
	local outputModel = RepStorage.OreModels:FindFirstChild(outputModelName)

	if not outputModel then
		warn("Output model '" .. outputModelName .. "' not found!"); processedItems[ITEM] = nil
		return true 
	end

	-- Trigger Sound on valid plate entry
	if CRUSH_SOUND and os.clock() - lastSoundTime > SOUND_COOLDOWN then
		lastSoundTime = os.clock()
		CRUSH_SOUND:Stop()
		CRUSH_SOUND.PlaybackSpeed = math.random(105, 125) / 100 -- Higher pitch for assembly clicks
		CRUSH_SOUND:Play()
	end

	ITEM_HANDLER.UPGRADE_ORE(PLOT, script.Parent.Parent, ITEM, ITEM_INFO.Addition, ITEM_INFO.Multiplier, ITEM_INFO.ItemId, 30)

	local newItem = outputModel:Clone()
	newItem.Name = ITEM.Name; CollectionService:AddTag(newItem, "MovingOre")
	newItem:SetAttribute("OreID", math.random(1, 1000000)); newItem:SetAttribute("LaneOffset", (math.random(-40, 40) / 100))
	Instance.new("StringValue", newItem).Name = "OreModelName"; newItem.OreModelName.Value = outputModelName

	if ITEM:FindFirstChild("Cash") then
		local val = Instance.new("NumberValue", newItem); val.Name = "Cash"; val.Value = ITEM.Cash.Value
	end

	for _, child in pairs(ITEM:GetChildren()) do
		if child:IsA("NumberValue") and child.Name ~= "Cash" then child:Clone().Parent = newItem end
	end

	local randomX = math.random(-SPREAD_WIDTH/2 * 100, SPREAD_WIDTH/2 * 100) / 100
	newItem:PivotTo(ASSEMBLY_OUTPUT.CFrame * CFrame.new(randomX, 0, 0))
	newItem.Parent = ITEM.Parent

	local p = (newItem:IsA("Model") and newItem.PrimaryPart) or newItem
	if p:IsA("BasePart") then
		p.Anchored = false
		local plotOwner = Players:FindFirstChild(PLOT.Owner.Value)
		if plotOwner then p:SetNetworkOwner(plotOwner) end
	end

	ITEM:Destroy()
	task.delay(1, function() processedItems[ITEM] = nil end)
	return true
end

-- ── DETECTION & INITIALIZATION ───────────────────────────────────────────
ASSEMBLY_CONFIG.Changed:Connect(updateDisplay)
updateDisplay()

task.spawn(function()
	local overlapParams = OverlapParams.new()
	overlapParams.FilterType = Enum.RaycastFilterType.Exclude
	overlapParams.FilterDescendantsInstances = {MODEL}

	while true do
		task.wait(DETECT_INTERVAL)
		local parts = workspace:GetPartBoundsInBox(ASSEMBLY_INPUT.CFrame, ASSEMBLY_INPUT.Size, overlapParams)

		for _, hit in pairs(parts) do
			local ITEM = nil

			-- Check if the part itself is the ore
			if hit:FindFirstChild("Cash") then
				ITEM = hit
				-- Otherwise check the parent (checking hit.Parent existence first)
			elseif hit.Parent and hit.Parent:FindFirstChild("Cash") then
				ITEM = hit.Parent
			end

			if ITEM and not processedItems[ITEM] then
				assembleItem(ITEM)
			end
		end
	end
end)

print("Plate Processor Scanning Logic Loaded!")
