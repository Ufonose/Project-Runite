-- ServerScriptService/Items/RollingMill.lua
local Players = game:GetService("Players")
local RepStorage = game:GetService("ReplicatedStorage")
local CollectionService = game:GetService("CollectionService")

local ITEM_HANDLER = require(RepStorage.Modules.ItemHandler)
local ITEM_INFO = require(script.Parent.Parent.ItemStats)
local PLOT = script.Parent.Parent.Parent.Parent

local ROLL_INPUT = script.Parent:WaitForChild("RollInput")
local ROLL_OUTPUT = script.Parent:WaitForChild("RollOutput")

local SPREAD_WIDTH = 2.7
local DETECT_INTERVAL = 0.1 

local processedItems = {}

local function rollIngot(ITEM)
	if not ITEM or processedItems[ITEM] then return end
	processedItems[ITEM] = true

	-- 1. IDENTIFY THE ITEM
	-- Look for ANY tag that tells us what this is
	local modelNameTag = ITEM:FindFirstChild("OreModelName") or ITEM:FindFirstChild("IngotType")
	local itemName = modelNameTag and modelNameTag.Value or ITEM.Name

	-- 2. THE BLACK HOLE RULE (Destroy everything that isn't an Ingot)
	if not string.find(itemName, "Ingot") then
		print("ROLLING MILL: Item is not an ingot, destroying: " .. itemName)
		ITEM:Destroy()
		return
	end

	-- 3. PREPARE THE PLATE
	local oreFolder = RepStorage:FindFirstChild("OreModels")
	local plateModelName = string.gsub(itemName, "Ingot", "Plate")
	local plateModel = oreFolder and oreFolder:FindFirstChild(plateModelName)

	if not plateModel then
		warn("ROLLING MILL: Plate model '" .. plateModelName .. "' not found! Destroying.")
		ITEM:Destroy()
		return
	end

	-- 4. UPGRADE & CLONE
	ITEM_HANDLER.UPGRADE_ORE(PLOT, script.Parent.Parent, ITEM, ITEM_INFO.Addition, ITEM_INFO.Multiplier, ITEM_INFO.ItemId, 30)

	local newPlate = plateModel:Clone()
	newPlate.Name = itemName .. "_Plate"

	-- Ensure the conveyor script sees it
	CollectionService:AddTag(newPlate, "MovingOre")
	newPlate:SetAttribute("OreID", math.random(1, 1000000))
	newPlate:SetAttribute("LaneOffset", (math.random(-5, 5) / 100))

	-- Metadata Tags
	local newPlateTag = Instance.new("StringValue", newPlate)
	newPlateTag.Name = "OreModelName"
	newPlateTag.Value = plateModelName

	if ITEM:FindFirstChild("Cash") then
		local cashValue = Instance.new("NumberValue", newPlate)
		cashValue.Name = "Cash"
		cashValue.Value = ITEM.Cash.Value
	end

	-- 5. POSITION & PHYSICS
	local randomX = math.random(-SPREAD_WIDTH/2 * 100, SPREAD_WIDTH/2 * 100) / 100
	newPlate:PivotTo(ROLL_OUTPUT.CFrame * CFrame.new(randomX, 0, 0))
	newPlate.Parent = ITEM.Parent

	local p = (newPlate:IsA("Model") and newPlate.PrimaryPart) or newPlate
	if p:IsA("BasePart") then
		p.Anchored = false
		p.CanCollide = true
		-- Hand control to the player immediately
		local plotOwner = Players:FindFirstChild(PLOT.Owner.Value)
		if plotOwner then p:SetNetworkOwner(plotOwner) end
	end

	-- 6. FINAL CLEANUP
	ITEM:Destroy()
	task.delay(1, function() processedItems[ITEM] = nil end)
end

-- ============================================================
-- THE DETECTION LOOP
-- ============================================================
task.spawn(function()
	local overlapParams = OverlapParams.new()
	-- We remove the Include filter to ensure it sees EVERYTHING that hits the box
	overlapParams.FilterType = Enum.RaycastFilterType.Exclude
	overlapParams.FilterDescendantsInstances = {script.Parent} -- Don't detect the machine itself

	while true do
		task.wait(DETECT_INTERVAL)

		-- Find every part inside the input box
		local partsInInput = workspace:GetPartBoundsInBox(ROLL_INPUT.CFrame, ROLL_INPUT.Size, overlapParams)

		for _, hit in pairs(partsInInput) do
			local ITEM = nil
			-- Find the "Ore" object by looking for the Cash value
			if hit:FindFirstChild("Cash") then
				ITEM = hit
			elseif hit.Parent and hit.Parent:FindFirstChild("Cash") then
				ITEM = hit.Parent
			end

			if ITEM and not processedItems[ITEM] then
				rollIngot(ITEM)
			end
		end
	end
end)
